<script>
  // --- HARDCODED CREDENTIALS ---
  const USER_CREDS = { username: "Miachen", password: "GJE8AT2021$" };
  const ADMIN_CREDS = { username: "Admin", password: "Admin123" };

  // --- HARDCODED CRYPTO PRICES ---
  const PRICES = { btc: 65000, eth: 3500, usdt: 1 };
  const NAMES = { btc: 'Bitcoin', eth: 'Ethereum', usdt: 'Tether' };

  // --- STATE MANAGEMENT ---
  // Using new localStorage keys to ensure you see the updated $10,300 balance
  const STORAGE_KEY_BALANCES = 'walletBalances_10300';
  const STORAGE_KEY_HISTORY = 'walletHistory_10300';

  // 0.15846154 BTC * $65,000 = $10,300.00 exactly
  const defaultBalances = { btc: 0.15846154, eth: 0, usdt: 0 };
  let currentSelectedAsset = 'btc'; 

  function getBalances() {
    const saved = localStorage.getItem(STORAGE_KEY_BALANCES);
    return saved ? JSON.parse(saved) : defaultBalances;
  }

  function getHistory() {
    const saved = localStorage.getItem(STORAGE_KEY_HISTORY);
    if (saved) return JSON.parse(saved);
    
    // Default fake deposit history so it totals $10,300 exactly. 
    return [
      { asset: 'btc', type: 'Deposit', change: 0.00307692, date: '06/28/2026' }, // ~$200 USD
      { asset: 'btc', type: 'Deposit', change: 0.07692308, date: '07/02/2026' }, // ~$5,000 USD
      { asset: 'btc', type: 'Deposit', change: 0.07846154, date: '07/08/2026' }  // ~$5,100 USD
    ];
  }

  // --- NAVIGATION LOGIC ---
  function showView(viewId) {
    document.querySelectorAll('.view-section').forEach(el => el.classList.remove('active'));
    document.getElementById(viewId).classList.add('active');
  }

  function checkLogin() {
    const userVal = document.getElementById("username").value;
    const passVal = document.getElementById("password").value;

    if (userVal === USER_CREDS.username && passVal === USER_CREDS.password) {
      loadUserWallet();
      showView('user-view');
      document.getElementById("username").value = "";
      document.getElementById("password").value = "";
    } 
    else if (userVal === ADMIN_CREDS.username && passVal === ADMIN_CREDS.password) {
      loadAdminDashboard();
      showView('admin-view');
      document.getElementById("username").value = "";
      document.getElementById("password").value = "";
    } 
    else {
      alert("Invalid credentials. Please try again.");
    }
  }

  function logout() {
    showView('login-view');
  }

  // --- USER WALLET LOGIC ---
  function loadUserWallet() {
    const balances = getBalances();
    
    const btcUsd = balances.btc * PRICES.btc;
    const ethUsd = balances.eth * PRICES.eth;
    const usdtUsd = balances.usdt * PRICES.usdt;
    const totalUsd = btcUsd + ethUsd + usdtUsd;

    const formatUsd = (num) => '$' + num.toLocaleString('en-US', {minimumFractionDigits: 2, maximumFractionDigits: 2});

    document.getElementById('display-total-usd').innerText = formatUsd(totalUsd);
    
    document.getElementById('user-btc-amt').innerText = `${balances.btc} BTC`;
    document.getElementById('user-btc-usd').innerText = formatUsd(btcUsd);

    document.getElementById('user-eth-amt').innerText = `${balances.eth} ETH`;
    document.getElementById('user-eth-usd').innerText = formatUsd(ethUsd);

    document.getElementById('user-usdt-amt').innerText = `${balances.usdt} USDT`;
    document.getElementById('user-usdt-usd').innerText = formatUsd(usdtUsd);
  }

  // --- CRYPTO DETAILS VIEW LOGIC ---
  function openCryptoDetails(assetKey) {
    currentSelectedAsset = assetKey;
    const balances = getBalances();
    const history = getHistory();
    
    const tokenAmt = balances[assetKey];
    const fiatAmt = tokenAmt * PRICES[assetKey];

    document.getElementById('details-header-title').innerText = NAMES[assetKey];
    document.getElementById('details-crypto-amt').innerText = `${tokenAmt} ${assetKey.toUpperCase()}`;
    document.getElementById('details-fiat-amt').innerText = '$' + fiatAmt.toLocaleString('en-US', {minimumFractionDigits: 2});

    const historyContainer = document.getElementById('details-history-list');
    historyContainer.innerHTML = '';

    const assetHistory = history.filter(item => item.asset === assetKey);

    if (assetHistory.length === 0) {
      historyContainer.innerHTML = '<div style="color:#8b949e; text-align:center; font-size:14px; padding-top:20px;">No transfers listed yet.</div>';
    } else {
      assetHistory.reverse().forEach(item => {
        const itemEl = document.createElement('div');
        itemEl.className = 'history-item';
        
        const isPos = item.change >= 0;
        const prefix = isPos ? '+' : '';
        const classColor = isPos ? 'positive' : 'negative';

        itemEl.innerHTML = `
          <div>
            <div class="history-type">${item.type}</div>
            <div class="history-date">${item.date}</div>
          </div>
          <div class="history-amount ${classColor}">
            ${prefix}${item.change} ${assetKey.toUpperCase()}
          </div>
        `;
        historyContainer.appendChild(itemEl);
      });
    }

    showView('details-view');
  }

  // --- WITHDRAWAL WINDOW INTERFACES ---
  function openGlobalWithdraw() {
    currentSelectedAsset = 'usdt'; 
    openAssetWithdraw();
  }

  function openAssetWithdraw() {
    const balances = getBalances();
    document.getElementById('withdraw-asset').value = NAMES[currentSelectedAsset] + ` (${currentSelectedAsset.toUpperCase()})`;
    document.getElementById('withdraw-max-label').innerText = `Available: ${balances[currentSelectedAsset]} ${currentSelectedAsset.toUpperCase()}`;
    document.getElementById('withdraw-address').value = '';
    document.getElementById('withdraw-amount').value = '';
    showView('withdraw-view');
  }

  function goBackFromWithdraw() {
    openCryptoDetails(currentSelectedAsset);
  }

  function processWithdrawal() {
    const address = document.getElementById('withdraw-address').value.trim();
    const amount = parseFloat(document.getElementById('withdraw-amount').value);
    const balances = getBalances();

    if (!address) {
      alert("Please enter a valid recipient network address.");
      return;
    }
    if (isNaN(amount) || amount <= 0) {
      alert("Please specify a valid quantity greater than zero.");
      return;
    }
    if (amount > balances[currentSelectedAsset]) {
      alert("Insufficient funds to complete this external settlement.");
      return;
    }

    balances[currentSelectedAsset] -= amount;
    localStorage.setItem(STORAGE_KEY_BALANCES, JSON.stringify(balances));

    const currentHistory = getHistory();
    currentHistory.push({
      asset: currentSelectedAsset,
      type: 'Withdrawal',
      change: -amount,
      date: new Date().toLocaleDateString()
    });
    localStorage.setItem(STORAGE_KEY_HISTORY, JSON.stringify(currentHistory));

    alert(`Transfer of ${amount} ${currentSelectedAsset.toUpperCase()} dispatched successfully.`);
    openCryptoDetails(currentSelectedAsset);
  }

  // --- ADMIN DASHBOARD LOGIC ---
  function loadAdminDashboard() {
    const balances = getBalances();
    document.getElementById('admin-btc').value = balances.btc;
    document.getElementById('admin-eth').value = balances.eth;
    document.getElementById('admin-usdt').value = balances.usdt;
  }

  function saveAdminBalances() {
    const oldBalances = getBalances();
    
    const newBtc = parseFloat(document.getElementById('admin-btc').value) || 0;
    const newEth = parseFloat(document.getElementById('admin-eth').value) || 0;
    const newUsdt = parseFloat(document.getElementById('admin-usdt').value) || 0;

    const newBalances = { btc: newBtc, eth: newEth, usdt: newUsdt };
    localStorage.setItem(STORAGE_KEY_BALANCES, JSON.stringify(newBalances));

    const currentHistory = getHistory();
    const assets = ['btc', 'eth', 'usdt'];
    let updatedAny = false;

    assets.forEach(asset => {
      const diff = newBalances[asset] - oldBalances[asset];
      if (diff !== 0) {
        updatedAny = true;
        currentHistory.push({
          asset: asset,
          type: diff > 0 ? 'Admin Deposit' : 'Admin Charge',
          change: parseFloat(diff.toFixed(8)), 
          date: new Date().toLocaleDateString()
        });
      }
    });

    if (updatedAny) {
      localStorage.setItem(STORAGE_KEY_HISTORY, JSON.stringify(currentHistory));
    }

    alert("User account configurations updated.");
  }
</script>
